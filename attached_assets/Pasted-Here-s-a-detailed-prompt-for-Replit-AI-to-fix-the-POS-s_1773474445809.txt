```javascript
// BULLETPROOF SIDEBAR MANAGER
class SidebarManager {
  constructor() {
    this.sidebar = document.querySelector('.sidebar, [data-sidebar]');
    this.ensureAlwaysVisible();
    this.bindPosEvents();
  }
  
  ensureAlwaysVisible() {
    // Run every 100ms during POS to prevent CSS overrides
    const checkSidebar = () => {
      this.sidebar.style.cssText = `
        position: fixed !important;
        left: 0 !important;
        z-index: 9999 !important;
        display: block !important;
        visibility: visible !important;
        opacity: 1 !important;
        pointer-events: auto !important;
      `;
      
      // Ensure main content respects sidebar
      document.querySelector('.main-content')?.style.setProperty('margin-left', '280px', 'important');
    };
    
    // Initial fix + watch for POS changes
    checkSidebar();
    const observer = new MutationObserver(checkSidebar);
    observer.observe(document.body, { attributes: true, childList: true });
  }
  
  bindPosEvents() {
    // Intercept POS activation
    document.addEventListener('click', (e) => {
      if (e.target.matches('[data-tab="pos"], .pos-tab')) {
        setTimeout(() => this.forceSidebarVisible(), 10);
      }
    });
  }
  
  forceSidebarVisible() {
    this.sidebar.classList.remove('hidden', 'collapsed', 'invisible');
    this.sidebar.style.display = 'block';
  }
}

// Initialize on load
document.addEventListener('DOMContentLoaded', () => {
  new SidebarManager();
});
```
