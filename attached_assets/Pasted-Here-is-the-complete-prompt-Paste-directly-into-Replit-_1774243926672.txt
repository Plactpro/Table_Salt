#### 3. Price resolution engine

Build a **PriceResolutionService** that finds the correct price for any item in any context:

```typescript
interface PriceContext {
  tenantId: string;
  outletId: string;
  menuItemId: string;
  orderType: 'DINE_IN' | 'TAKEAWAY' | 'DELIVERY' | 'QR_ORDER';
  orderTime: Date;
  customerId?: string;
  customerSegment?: string;
}

async function resolvePrice(context: PriceContext): Promise<ResolvedPrice> {

  // Step 1: Get all active price rules for this item + outlet
  const rules = await getPriceRules({
    tenantId: context.tenantId,
    outletId: context.outletId,
    menuItemId: context.menuItemId,
    isActive: true
  });

  // Step 2: Filter rules that match current context
  const matchingRules = rules.filter(rule =>
    matchesOrderType(rule, context.orderType) &&
    matchesTimeSlot(rule, context.orderTime) &&
    matchesDayOfWeek(rule, context.orderTime) &&
    matchesCustomerSegment(rule, context.customerSegment) &&
    matchesDateRange(rule, context.orderTime)
  );

  // Step 3: Sort by priority (highest first)
  matchingRules.sort((a, b) => b.priority - a.priority);

  // Step 4: Apply highest priority matching rule
  if (matchingRules.length > 0) {
    const bestRule = matchingRules[0];
    return {
      price: bestRule.price,
      ruleId: bestRule.id,
      priceType: bestRule.priceType,
      reason: buildResolutionReason(bestRule, context)
    };
  }

  // Step 5: Fallback to outlet base price
  const outletBase = await getOutletBasePrice(
    context.outletId,
    context.menuItemId
  );
  if (outletBase) return { price: outletBase, priceType: 'OUTLET_BASE' };

  // Step 6: Final fallback to global menu base price
  const globalBase = await getMenuItemBasePrice(context.menuItemId);
  return { price: globalBase, priceType: 'GLOBAL_BASE' };
}
```

##### Price resolution priority order:
```
1. Event pricing (highest — overrides everything)
2. Customer segment pricing
3. Time slot + day combination pricing
4. Time slot pricing only
5. Day-based pricing only
6. Order type pricing
7. Outlet base price
8. Global menu base price (lowest fallback)
```

#### 8. Price conflict detection and warnings

```typescript
// When saving a new price rule, check for conflicts
async function detectPriceConflicts(
  newRule: PriceRule
): Promise<Conflict[]> {

  const conflicts = [];

  // Find existing rules with same conditions
  const overlapping = await findOverlappingRules(newRule);

  for (const existing of overlapping) {
    if (existing.priority === newRule.priority) {
      conflicts.push({
        type: 'SAME_PRIORITY',
        message: `Rule "${existing.notes}" has same priority.
                  Increase priority to override it.`,
        existingRule: existing
      });
    }

    if (newRule.price < getCostPrice(newRule.menuItemId)) {
      conflicts.push({
        type: 'BELOW_COST',
        message: `Price ₹${newRule.price} is below cost
                  price ₹${costPrice}. Selling at a loss.`,
        severity: 'warning'
      });
    }
  }

  return conflicts;
}
```
