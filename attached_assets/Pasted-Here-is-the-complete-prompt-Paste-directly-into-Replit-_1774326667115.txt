#### 1. Plan restriction — Enterprise only

```typescript
// Middleware: check plan before any ad feature access
async function requireEnterprisePlan(
  req: Request,
  res: Response,
  next: NextFunction
): Promise<void> {

  const subscription = await getSubscription(req.tenantId);

  if (subscription.plan_name !== 'ENTERPRISE') {
    return res.status(403).json({
      error: 'AD_FEATURE_RESTRICTED',
      message: 'Advertisement display is available '
               + 'on Enterprise plan only',
      upgradeUrl: '/settings/billing/upgrade',
      currentPlan: subscription.plan_name
    });
  }

  next();
}

#### 2. Core concept

```
TWO AD TYPES:

TYPE 1: OWN ADS (Restaurant's own content)
  → Promote their own dishes, offers, events
  → No revenue sharing
  → Full control by restaurant owner
  → Free to display

TYPE 2: THIRD PARTY ADS (Revenue generation)
  → External businesses advertise on screen
  → Restaurant earns per display / per day
  → Super admin approves all 3rd party ads
  → Revenue tracked and reported monthly

DISPLAY LOCATIONS:
  → Kiosk screen (primary)
  → Digital menu board (KDS wall screen idle mode)
  → QR ordering page (banner ads)
  → Billing/payment waiting screen

REVENUE MODEL (3rd party):
  → Per day rate: advertiser pays X per day
  → Per display rate: pays per 1000 impressions
  → Time slot rate: pays for specific hours
```

#### 4. File size restrictions — strict enforcement

```typescript
// File size and type restrictions
const AD_FILE_RESTRICTIONS = {

  IMAGE: {
    maxSizeBytes: 2 * 1024 * 1024,  // 2 MB max
    maxSizeDisplay: '2 MB',
    allowedTypes: ['image/jpeg', 'image/png',
                   'image/webp', 'image/gif'],
    allowedExtensions: ['.jpg', '.jpeg',
                        '.png', '.webp', '.gif'],
    minDimensions: { width: 800, height: 450 },
    maxDimensions: { width: 1920, height: 1080 },
    recommendedDimensions: '1920×1080 (Full HD)'
  },

  VIDEO: {
    maxSizeBytes: 50 * 1024 * 1024,  // 50 MB max
    maxSizeDisplay: '50 MB',
    allowedTypes: ['video/mp4', 'video/webm'],
    allowedExtensions: ['.mp4', '.webm'],
    maxDurationSeconds: 30,
    maxResolution: '1920x1080',
    recommendedFormat: 'MP4, H.264, max 30 seconds'
  },

  HTML_BANNER: {
    maxSizeBytes: 512 * 1024,  // 512 KB max
    maxSizeDisplay: '512 KB',
    allowedTypes: ['text/html'],
    allowedExtensions: ['.html'],
    noExternalScripts: true,
    noIframes: true
  },

  QR_BANNER: {
    // Banner on QR ordering page
    maxSizeBytes: 1 * 1024 * 1024,  // 1 MB max
    maxSizeDisplay: '1 MB',
    allowedTypes: ['image/jpeg', 'image/png',
                   'image/webp'],
    recommendedDimensions: '1200×300 (banner)'
  }
};

// Validation function
async function validateAdFile(
  file: UploadedFile,
  adType: string
): Promise<ValidationResult> {

  const restrictions = AD_FILE_RESTRICTIONS[adType];
  const errors = [];
  const warnings = [];

  // Check file size
  if (file.size > restrictions.maxSizeBytes) {
    errors.push(
      `File too large: ${formatBytes(file.size)}. `
      + `Maximum allowed: ${restrictions.maxSizeDisplay}`
    );
  }

  // Check file type
  if (!restrictions.allowedTypes.includes(file.mimetype)) {
    errors.push(
      `File type not allowed: ${file.mimetype}. `
      + `Allowed: ${restrictions.allowedExtensions.join(', ')}`
    );
  }

  // Check image dimensions
  if (adType === 'IMAGE') {
    const dimensions = await getImageDimensions(file);
    if (dimensions.width < restrictions.minDimensions.width
        || dimensions.height < restrictions.minDimensions.height) {
      warnings.push(
        `Image resolution too low: ${dimensions.width}×${dimensions.height}. `
        + `Minimum: ${restrictions.minDimensions.width}×${restrictions.minDimensions.height}`
      );
    }
  }

  // Check video duration
  if (adType === 'VIDEO') {
    const duration = await getVideoDuration(file);
    if (duration > restrictions.maxDurationSeconds) {
      errors.push(
        `Video too long: ${duration}s. `
        + `Maximum: ${restrictions.maxDurationSeconds}s`
      );
    }
  }

  return {
    valid: errors.length === 0,
    errors,
    warnings,
    fileInfo: {
      size: formatBytes(file.size),
      type: file.mimetype,
      name: file.originalname
    }
  };
}
