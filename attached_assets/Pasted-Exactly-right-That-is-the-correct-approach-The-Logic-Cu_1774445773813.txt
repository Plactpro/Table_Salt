#### 4. Jurisdiction resolver service

```typescript
class JurisdictionService {

  // Called when currency is saved in outlet settings
  async resolveJurisdiction(
    outletId: string,
    currencyCode: string
  ): Promise<JurisdictionConfig> {

    const jurisdiction =
      CURRENCY_JURISDICTION_MAP[currencyCode]
      ?? CURRENCY_JURISDICTION_MAP['USD'];
      // fallback to US rules for unknown currencies

    // Auto-update outlet jurisdiction
    await db.outlet_settings.update({
      where: { outletId },
      data: {
        jurisdiction_code: currencyCode,
        tax_rate: jurisdiction.defaultTaxRate
        // only set if not already configured
      }
    });

    return jurisdiction;
  }

  // Called everywhere to get current rules
  async getJurisdictionConfig(
    outletId: string
  ): Promise<JurisdictionConfig> {

    const outlet = await getOutletSettings(outletId);
    return CURRENCY_JURISDICTION_MAP[
      outlet.currency_code
    ] ?? CURRENCY_JURISDICTION_MAP['USD'];
  }
}
```

      bill.total,
      jurisdiction.roundingRule

    regulatoryFooter:
