**Live Time Zone Clock:**
```javascript
function updateTimePreview() {
  const now = new Date().toLocaleString('en-US', {
    timeZone: APP_CONFIG.timezone.zone,
    hour12: APP_CONFIG.timezone.format === '12hr',
    weekday: 'short',
    year: 'numeric',
    month: 'short',
    day: 'numeric',
    hour: '2-digit',
    minute: '2-digit'
  });
  document.getElementById('time-preview').textContent = now;
}
setInterval(updateTimePreview, 1000);
```

function generateReceiptPreview(subtotal = 150) {

### **9. Persistence & App Integration**

**Save/Load:**
```javascript
async function saveAllConfig() {
  localStorage.setItem('app_config', JSON.stringify(APP_CONFIG));
  
  // Trigger app-wide refresh
  window.dispatchEvent(new CustomEvent('config-updated'));
  
  // POS, Billing, Reports auto-update
  updateAllDisplays();
}

function loadConfig() {
  const saved = localStorage.getItem('app_config');
  if (saved) Object.assign(APP_CONFIG, JSON.parse(saved));
  
  // Apply to UI
  populateFormFromConfig();
}
```
