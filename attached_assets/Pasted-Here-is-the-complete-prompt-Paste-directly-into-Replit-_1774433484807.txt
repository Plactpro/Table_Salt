##### 3.2 Read-only enforcement middleware

```typescript
// Middleware applied to ALL routes during impersonation
async function enforceReadOnlyMode(
  req: Request,
  res: Response,
  next: NextFunction
): Promise<void> {

  const session = await getActiveImpersonationSession(
    req.sessionToken
  );

  if (!session) return next();
  if (session.access_mode !== 'READ_ONLY') return next();

  // Block all write operations
  const writeMethods = ['POST', 'PUT', 'PATCH', 'DELETE'];

  if (writeMethods.includes(req.method)) {

    // Log blocked attempt
    await logImpersonationAction(session.id, {
      action_type: 'EDIT_ATTEMPTED',
      action_description:
        `Blocked: attempted ${req.method} to ${req.path}`,
      page_url: req.path,
      was_blocked: true
    });

    return res.status(403).json({
      error: 'READ_ONLY_SESSION',
      message: 'This impersonation session is read-only. '
               + 'Unlock edit mode to make changes.',
      canUnlock: true
    });
  }

  // Log page view
  await logImpersonationAction(session.id, {
    action_type: 'PAGE_VIEW',
    action_description: `Viewed: ${req.path}`,
    page_url: req.path
  });

  next();
}
```
