#### 2. Sound files — Web Audio API fallback

Use actual audio files where possible. If files are not loaded, use Web Audio API tones as fallback so alerts always work:

```typescript
class SoundPlayer {

  private audioContext: AudioContext;

  // Fallback tones if MP3 files not available
  private fallbackTones: Record<string, ToneConfig> = {
    'new_order.mp3':       { freq: 880, pattern: [150, 80, 150] },
    'rush_order.mp3':      { freq: 660, pattern: [100, 50, 100, 50, 100] },
    'allergy_alarm.mp3':   { freq: 440, pattern: [300, 100, 300, 100, 500] },
    'order_ready.mp3':     { freq: 1047, pattern: [200, 100, 200, 100, 400] },
    'overdue_warning.mp3': { freq: 523, pattern: [200, 100, 200] },
    'waiter_call.mp3':     { freq: 784, pattern: [150, 100, 150] },
    'printer_error.mp3':   { freq: 330, pattern: [500, 200, 500] },
    'attention_chime.mp3': { freq: 659, pattern: [200, 150, 200] },
    'stock_alert.mp3':     { freq: 494, pattern: [300, 150, 300] },
    'urgent_tone.mp3':     { freq: 392, pattern: [100, 50, 100, 50, 300] },
    'reminder_chime.mp3':  { freq: 698, pattern: [200, 200, 200] }
  };

  async play(soundFile: string, volume: number): Promise<void> {
    try {
      // Try MP3 first
      await this.playMP3(soundFile, volume);
    } catch {
      // Fallback to generated tone
      const tone = this.fallbackTones[soundFile];
      if (tone) await this.playTone(tone, volume);
    }
  }

  private async playTone(
    config: ToneConfig,
    volume: number
  ): Promise<void> {
    const ctx = this.audioContext;
    const gainNode = ctx.createGain();
    gainNode.gain.value = volume / 100;
    gainNode.connect(ctx.destination);

    let time = ctx.currentTime;
    for (let i = 0; i < config.pattern.length; i++) {
      if (i % 2 === 0) {
        // ON: play tone
        const osc = ctx.createOscillator();
        osc.frequency.value = config.freq;
        osc.connect(gainNode);
        osc.start(time);
        osc.stop(time + config.pattern[i] / 1000);
      }
      time += config.pattern[i] / 1000;
    }
  }
}
```

#### 3. Alert engine service (simplified)

```typescript
class AlertEngine {

  async trigger(
    alertCode: string,
    context: {
      outletId: string;
      referenceId?: string;
      referenceNumber?: string;
      message: string;
      specificUserIds?: string[];
    }
  ): Promise<void> {

    // Get definition
    const def = await getAlertDefinition(alertCode);
    if (!def?.is_active) return;

    // Get outlet config (volume, enabled/disabled)
    const config = await getOutletConfig(
      context.outletId, alertCode
    );

    // Respect disable setting (except allergy)
    if (config && !config.is_enabled
        && def.can_be_disabled) return;

    // Resolve which users to alert
    const users = context.specificUserIds
      ? await getUsersByIds(context.specificUserIds)
      : await getOnlineUsersByRoles(
          context.outletId,
          def.target_roles
        );

    // Volume: use outlet config or default 80%
    const volume = Math.max(
      config?.volume_level ?? 80,
      def.min_volume  // allergy enforces minimum
    );

    // Save alert event
    const event = await db.alert_events.create({
      alertCode,
      urgency: def.urgency,
      referenceId: context.referenceId,
      referenceNumber: context.referenceNumber,
      message: context.message,
      outletId: context.outletId,
      targetUserIds: users.map(u => u.id)
    });

    // Deliver via WebSocket
    for (const user of users) {
      socket.to(`user_${user.id}`).emit('alert', {
        eventId: event.id,
        alertCode,
        sound: config?.custom_sound ?? def.sound_file,
        volume,
        urgency: def.urgency,
        message: context.message,
        referenceId: context.referenceId,
        requiresAcknowledge: def.requires_acknowledge
      });
    }

    // Schedule repeat if needed
    if (def.repeat_interval_sec > 0) {
      await scheduleRepeat(event.id, def);
    }
  }

  async acknowledge(
    eventId: string,
    userId: string
  ): Promise<void> {
    await db.alert_events.update(eventId, {
      acknowledged_by: userId,
      acknowledged_at: new Date(),
      is_resolved: true
    });
    await cancelRepeat(eventId);
  }
}
```
