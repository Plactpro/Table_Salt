### **Phase 3: Single Source of Truth Functionality**

**Unified Contact Sales Handler:**
```javascript
class ContactSalesManager {
  constructor() {
    this.primaryButton = '.btn-enterprise-sales';
    this.salesForm = '#contact-sales-modal';
    this.init();
  }
  
  init() {
    // ONLY bind to Enterprise button
    document.addEventListener('click', (e) => {
      if (e.target.matches(this.primaryButton, '[data-trigger="contact-sales"]')) {
        e.preventDefault();
        this.openSalesModal();
      }
    });
    
    // Hide ALL floating duplicates
    this.hideFloatingDuplicates();
  }
  
  hideFloatingDuplicates() {
    const duplicates = document.querySelectorAll(`
      .btn-sales-floating,
      .contact-sales-fixed,
      .gold-sales-btn,
      [class*="sales"][class*="float"],
      [class*="contact"][class*="fixed"]
    `);
    duplicates.forEach(btn => btn.remove());
  }
  
  openSalesModal() {
    // Show enhanced modal with Enterprise context
    const modalData = {
      source: 'enterprise_button',
      planInterest: 'Enterprise',
      timestamp: new Date().toISOString()
    };
    showContactSalesForm(modalData);
  }
}
```
