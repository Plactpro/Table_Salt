### Special Resource Management System (High Chairs, Baby Seats & More)

This is a continuation of the existing restaurant SaaS project in Replit. The following are already built: multi-tenant architecture, POS, table management, floor plan, QR customer requests, reservation system, outlet settings, and service coordination.

Now build a **fully functional Special Resource Management System** that tracks availability of high chairs, baby seats, booster seats, wheelchair access, and other special resources — showing real-time availability to staff, allowing pre-booking via reservations, and alerting when resources run out.

***

#### 1. Core concept

```
OWNER sets up resources in outlet settings
(how many high chairs, booster seats, etc.)
        ↓
Resources tracked in real time:
  Available / In Use / Under Cleaning / Damaged
        ↓
Staff assigns resource when:
  → Walk-in customer requests
  → Table is assigned to customer
  → Reservation pre-books resource
  → Customer requests via QR
        ↓
System shows:
  → Floor plan: which table has which resource
  → Dashboard: resource availability count
  → Reservation screen: pre-booked resources
        ↓
Alerts when:
  → All resources of a type are in use
  → Reservation with resource need arriving soon
  → Resource marked damaged reduces availability
```

#### 2. Database schema

```sql
-- Resource types master (configured per outlet)
CREATE TABLE special_resources (
  id                    UUID PRIMARY KEY,
  tenant_id             UUID NOT NULL,
  outlet_id             UUID NOT NULL,
  resource_code         VARCHAR(30) NOT NULL,
    -- 'HIGH_CHAIR', 'BOOSTER_SEAT', 'BABY_COT',
    -- 'WHEELCHAIR', 'PRAYER_MAT', 'WALKING_FRAME',
    -- 'CUSTOM'
  resource_name         VARCHAR(100) NOT NULL,
    -- display name: 'High Chair', 'Baby Seat'
  resource_icon         VARCHAR(10) DEFAULT '🪑',
  total_units           INT NOT NULL DEFAULT 0,
    -- total owned by outlet
  available_units       INT NOT NULL DEFAULT 0,
    -- currently available for use
  in_use_units          INT DEFAULT 0,
  under_cleaning_units  INT DEFAULT 0,
  damaged_units         INT DEFAULT 0,
  is_trackable          BOOLEAN DEFAULT true,
    -- false = unlimited (e.g. wheelchair ramps)
  requires_setup_time   INT DEFAULT 0,
    -- minutes needed to set up before use
  notes                 TEXT,
  is_active             BOOLEAN DEFAULT true,
  created_at            TIMESTAMP DEFAULT NOW(),
  updated_at            TIMESTAMP DEFAULT NOW()
);

-- Individual resource units
CREATE TABLE resource_units (
  id                    UUID PRIMARY KEY,
  tenant_id             UUID NOT NULL,
  outlet_id             UUID NOT NULL,
  resource_id           UUID NOT NULL,
  unit_code             VARCHAR(30) NOT NULL,
    -- 'HC-01', 'HC-02', 'BS-01' etc.
  unit_name             VARCHAR(100),
  status                VARCHAR(20) DEFAULT 'available',
    -- 'available', 'in_use', 'cleaning',
    -- 'damaged', 'reserved', 'retired'
  current_table_id      UUID,
  current_order_id      UUID,
  last_cleaned_at       TIMESTAMP,
  notes                 TEXT,
  created_at            TIMESTAMP DEFAULT NOW()
);

-- Resource assignments per table/order
CREATE TABLE resource_assignments (
  id                    UUID PRIMARY KEY,
  tenant_id             UUID NOT NULL,
  outlet_id             UUID NOT NULL,
  resource_id           UUID NOT NULL,
  resource_name         VARCHAR(100),
  resource_unit_id      UUID,
  unit_code             VARCHAR(30),
  table_id              UUID,
  table_number          VARCHAR(20),
  order_id              UUID,
  reservation_id        UUID,
  quantity              INT DEFAULT 1,
  assigned_for          VARCHAR(50),
    -- 'INFANT', 'TODDLER', 'ELDERLY',
    -- 'DISABLED', 'GENERAL'
  status                VARCHAR(20) DEFAULT 'assigned',
    -- 'reserved', 'assigned', 'in_use',
    -- 'returned', 'cleaning_needed'
  special_notes         TEXT,
  assigned_by           UUID,
  assigned_by_name      VARCHAR(255),
  assigned_at           TIMESTAMP DEFAULT NOW(),
  returned_at           TIMESTAMP,
  requires_cleaning     BOOLEAN DEFAULT false,
  created_at            TIMESTAMP DEFAULT NOW()
);

-- Resource cleaning log
CREATE TABLE resource_cleaning_log (
  id                    UUID PRIMARY KEY,
  tenant_id             UUID NOT NULL,
  resource_unit_id      UUID NOT NULL,
  unit_code             VARCHAR(30),
  resource_name         VARCHAR(100),
  cleaning_type         VARCHAR(20) DEFAULT 'STANDARD',
    -- 'STANDARD', 'DEEP_CLEAN', 'SANITIZE'
  started_at            TIMESTAMP DEFAULT NOW(),
  completed_at          TIMESTAMP,
  cleaned_by            UUID,
  cleaned_by_name       VARCHAR(255),
  notes                 TEXT
);
```
