├── T7: WAITER PICKUP TIME

#### 6. Bottleneck detection engine

```typescript
async function detectBottlenecks(
  outletId: string,
  dateRange: DateRange
): Promise<Bottleneck[]> {

  const bottlenecks = [];

  // Check 1: Waiter delay (slow to send KOT)
  const slowWaiters = await query(`
    SELECT waiter_id, waiter_name,
           AVG(waiter_response_time) as avg_response,
           COUNT(*) as order_count
    FROM order_time_summary
    WHERE outlet_id = $1
    AND waiter_response_time > 180  -- > 3 minutes
    GROUP BY waiter_id, waiter_name
    HAVING COUNT(*) > 5
    ORDER BY avg_response DESC
  `);

  // Check 2: Kitchen idle time (ticket not picked up)
  const slowPickup = await query(`
    SELECT counter_name,
           AVG(kitchen_pickup_time) as avg_pickup
    FROM item_time_logs
    WHERE outlet_id = $1
    AND kitchen_pickup_time > 120  -- > 2 minutes
    GROUP BY counter_name
  `);

  // Check 3: Slow dishes (consistently over recipe time)
  const slowDishes = await query(`
    SELECT menu_item_name,
           AVG(actual_cooking_time) as avg_time,
           recipe_estimated_time,
           AVG(variance_percent) as avg_variance
    FROM item_time_logs
    WHERE outlet_id = $1
    AND variance_percent > 20  -- 20% over estimate
    GROUP BY menu_item_name, recipe_estimated_time
    HAVING COUNT(*) > 10
  `);

  // Check 4: Pass wait time (food sitting uncollected)
  const longPassWait = await query(`
    SELECT AVG(pass_wait_time) as avg_wait,
           MAX(pass_wait_time) as max_wait,
           DATE_TRUNC('hour', order_received_at) as hour
    FROM item_time_logs
    WHERE outlet_id = $1
    AND pass_wait_time > 300  -- > 5 min at pass
    GROUP BY hour
    ORDER BY avg_wait DESC
  `);

  return buildBottleneckReport(
    slowWaiters, slowPickup, slowDishes, longPassWait
  );
}
```

#### 8. Automated recipe time calibration

Update recipe estimated times based on real data:

```typescript
// Run weekly — recalibrates recipe prep time estimates
async function calibrateRecipeTimes(
  tenantId: string
): Promise<void> {

  const dishes = await getMenuItemsWithEnoughData(
    tenantId,
    minSampleCount: 20
  );

  for (const dish of dishes) {
    const stats = await getTimeStats(dish.menuItemId);

    // Use 75th percentile as new estimate
    // (achievable by most chefs, not just the fastest)
    const newEstimate = stats.percentile75;

    // Only update if variance > 15%
    if (Math.abs(newEstimate - dish.currentEstimate)
        / dish.currentEstimate > 0.15) {

      await updateRecipeEstimate(
        dish.menuItemId,
        newEstimate,
        `Auto-calibrated from ${stats.sampleCount} orders`
      );

      // Notify head chef of significant changes
      if (Math.abs(newEstimate - dish.currentEstimate) > 3) {
        await notifyHeadChef({
          message: `Recipe time updated: ${dish.name}
                    Old: ${dish.currentEstimate} min
                    New: ${newEstimate} min
                    Based on ${stats.sampleCount} orders`
        });
      }
    }
  }
}
```
